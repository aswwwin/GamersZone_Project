<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Database\Eloquent\Model;

class Game extends Model
{
    use HasFactory;
    
    protected $table = 'games';
    protected $primaryKey = 'game_id';
    
    protected $fillable = [
        'game_name',
        'description',
        'publisher',
        'genre',
        'release_date',
        'image_url',
        'is_popular',
        'is_active'
    ];
    
    // Define relationships
    public function inGameCurrencies()
    {
        return $this->hasMany(InGameCurrency::class, 'game_id', 'game_id');
    }
    
    public function coupons()
    {
        return $this->hasMany(Coupon::class, 'game_id', 'game_id');
    }
}
